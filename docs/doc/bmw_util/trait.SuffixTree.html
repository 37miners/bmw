fn tmatch(&mut self, text: &[u8], matches: &mut [Match]) -> Result<usize, Error>

return matches associated with the supplied text for this crate::SuffixTree. Matches are returned in the matches array supplied by the caller. The result is the number of matches found or a bmw_err::Error if an error occurs.
