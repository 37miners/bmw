Expand description

crate::LockBox is the same as crate::Lock except that it is possible to build The LockBox into a Box<dyn LockBox> structure so that it is object safe. It can then be cloned using DynClone.

Examples


 use bmw_err::*;
 use bmw_util::*;

 struct TestLockBox<T> {
     lock_box: Box<dyn LockBox<T>>,
 }

 fn test_lock_box() -> Result<(), Error> {
     let lock_box = lock_box!(1u32)?;
     let mut lock_box2 = lock_box.clone();
     let mut tlb = TestLockBox { lock_box };

     {
         let mut tlb = tlb.lock_box.wlock()?;
         (**tlb.guard()) = 2u32;
     }

     {
         let tlb = tlb.lock_box.rlock()?;
         assert_eq!((**tlb.guard()), 2u32);
     }

     Ok(())
 }
